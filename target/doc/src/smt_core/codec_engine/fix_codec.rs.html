11#[derive(Debug, thiserror::Error)]
12pub enum CodecError {
13    #[error("I/O error: {0}")]
14    Io(#[from] std::io::Error),
15    #[error("UTF-8 conversion error (string): {0}")]
16    StringUtf8Error(#[from] std::string::FromUtf8Error),
17    #[error("UTF-8 conversion error (str): {0}")]
18    StrUtf8Error(#[from] std::str::Utf8Error),
19    #[error("Incomplete message: more data needed")]
20    IncompleteMessage,
21    #[error("Invalid message format: {0}")]
22    InvalidFormat(String),
23    #[error("Missing required field: tag {0}")]
24    MissingField(u32),
25    #[error("Invalid value for tag {tag}: {value}")]
26    InvalidValue { tag: u32, value: String },
27    #[error("Unsupported message type: {0}")]
28    UnsupportedMessageType(String),
29    #[error("Checksum mismatch: expected {expected}, got {actual}")]
30    ChecksumMismatch { expected: u8, actual: u8 },
31    #[error("Parse decimal error: {0}")]
32    ParseDecimalError(#[from] rust_decimal::Error),
33    #[error("Parse int error: {0}")]
34    ParseIntError(#[from] std::num::ParseIntError),
35}

37fn calculate_checksum(buffer: &[u8]) -> u8 {
38    buffer.iter().fold(0u8, |acc, &x| acc.wrapping_add(x))
39}

41fn append_tag_value(buf: &mut BytesMut, tag: u32, value: &str) {
42    buf.put_slice(tag.to_string().as_bytes());
43    buf.put_u8(b'=');
44    buf.put_slice(value.as_bytes());
45    buf.put_u8(SOH);
46}
47
48fn append_tag_decimal(buf: &mut BytesMut, tag: u32, value: Decimal) {
49    append_tag_value(buf, tag, &value.to_string());
50}
51
52fn append_tag_u32(buf: &mut BytesMut, tag: u32, value: u32) {
53    append_tag_value(buf, tag, &value.to_string());
54}
55
56fn append_tag_char(buf: &mut BytesMut, tag: u32, value: char) {
57    append_tag_value(buf, tag, &value.to_string());
58}

73impl FixEncoder for FixNewOrderSingle {
74    fn msg_type(&self) -> &str { "D" }
75    fn encode_body(&self, body_buf: &mut BytesMut) -> Result<(), CodecError> {
76        append_tag_value(body_buf, 11, &self.cl_ord_id);
77        append_tag_value(body_buf, 55, self.symbol.as_str());
78        let side_char = match self.side {
79            Side::Buy => '1',
80            Side::Sell => '2',
81        };
82        append_tag_char(body_buf, 54, side_char);
83        append_tag_u32(body_buf, 60, self.transact_time as u32); // Assuming Timestamp is u64, cast to u32 for simplicity
84        append_tag_decimal(body_buf, 38, self.order_qty);
85        let ord_type_char = match self.ord_type {
86            OrderType::Market => '1',
87            OrderType::Limit => '2',
88        };
89        append_tag_char(body_buf, 40, ord_type_char);
90        if let Some(price) = self.price {
91            append_tag_decimal(body_buf, 44, price);
92        }
93        if let Some(tif) = self.tif {
94            let tif_char = match tif {
95                TimeInForce::Day => '0',
96                TimeInForce::GTC => '1',
97                TimeInForce::IOC => '3',
98                TimeInForce::FOK => '4',
99                TimeInForce::GTD => '6',
100            };
101            append_tag_char(body_buf, 59, tif_char);
102        }
103        Ok(())
104    }
105}

107// Helper to extract a required field
108fn get_field_str(fields: &mut std::collections::HashMap<u32, Vec<u8>>, tag: u32) -> Result<String, CodecError> {
109    fields.remove(&tag)
110        .ok_or(CodecError::MissingField(tag))
111        .and_then(|v| String::from_utf8(v).map_err(CodecError::from))
112}
113
114fn get_field_u32(fields: &mut std::collections::HashMap<u32, Vec<u8>>, tag: u32) -> Result<u32, CodecError> {
115    get_field_str(fields, tag)?.parse().map_err(CodecError::from)
116}
117
118fn get_field_decimal(fields: &mut std::collections::HashMap<u32, Vec<u8>>, tag: u32) -> Result<Decimal, CodecError> {
119    get_field_str(fields, tag)?.parse().map_err(CodecError::from)
120}
121
122fn get_field_char(fields: &mut std::collections::HashMap<u32, Vec<u8>>, tag: u32) -> Result<char, CodecError> {
123    let s = get_field_str(fields, tag)?;
124    s.chars().next().ok_or_else(|| CodecError::InvalidValue { tag, value: s.clone() })
125}

127impl FixMessageBodyDecoder for FixNewOrderSingle {
128    fn decode_body(fields: &mut std::collections::HashMap<u32, Vec<u8>>) -> Result<Self, CodecError> {
129        let cl_ord_id = get_field_str(fields, 11)?;
130        let symbol_str = get_field_str(fields, 55)?;
131        let symbol = Symbol::new(&symbol_str);
132        let side_char = get_field_char(fields, 54)?;
133        let side = match side_char {
134            '1' => Side::Buy,
135            '2' => Side::Sell,
136            _ => return Err(CodecError::InvalidValue{tag: 54, value: side_char.to_string()}),
137        };
138        let transact_time = get_field_u32(fields, 60)? as u64; // Assuming Timestamp is u64
139        let order_qty = get_field_decimal(fields, 38)?;
140        let ord_type_char = get_field_char(fields, 40)?;
141        let ord_type = match ord_type_char {
142            '1' => OrderType::Market,
143            '2' => OrderType::Limit,
144            _ => return Err(CodecError::InvalidValue{tag: 40, value: ord_type_char.to_string()}),
145        };
146        let price = fields.remove(&44)
147            .map(|v| String::from_utf8(v).map_err(CodecError::from).and_then(|s| s.parse().map_err(CodecError::from)))
148            .transpose()?;
149        let tif = fields.remove(&59)
150            .map(|v| String::from_utf8(v).map_err(CodecError::from).and_then(|s| {
151                let c = s.chars().next().ok_or_else(|| CodecError::InvalidValue{tag: 59, value: s.clone()})?;
152                match c {
153                    '0' => Ok(TimeInForce::Day),
154                    '1' => Ok(TimeInForce::GTC),
155                    '3' => Ok(TimeInForce::IOC),
156                    '4' => Ok(TimeInForce::FOK),
157                    '6' => Ok(TimeInForce::GTD),
158                    _ => Err(CodecError::InvalidValue{tag: 59, value: c.to_string()}),
159                }
160            }))
161            .transpose()?;
162
163        Ok(FixNewOrderSingle {
164            cl_ord_id,
165            symbol,
166            side,
167            transact_time,
168            order_qty,
169            ord_type,
170            price,
171            tif,
172        })
173    }
174}

203pub fn encode_fix_message<T: FixEncoder>(
204    msg_body: &T,
205    sender_comp_id: &str,
206    target_comp_id: &str,
207    msg_seq_num: u32,
208    sending_time: u64, // Simplified timestamp
209) -> Result<Bytes, CodecError> {
210    let mut body_buf = BytesMut::new();
211    msg_body.encode_body(&mut body_buf)?;
212
213    let mut header_buf = BytesMut::new();
214    append_tag_value(&mut header_buf, 8, "FIX.4.2"); // Example, should be configurable
215    append_tag_u32(&mut header_buf, 9, body_buf.len() as u32);
216    append_tag_value(&mut header_buf, 35, msg_body.msg_type());
217    append_tag_value(&mut header_buf, 49, sender_comp_id);
218    append_tag_value(&mut header_buf, 56, target_comp_id);
219    append_tag_u32(&mut header_buf, 34, msg_seq_num);
220    append_tag_u32(&mut header_buf, 52, sending_time as u32); // Simplified sending time
221
222    let mut full_msg_buf = BytesMut::new();
223    full_msg_buf.put_slice(&header_buf);
224    full_msg_buf.put_slice(&body_buf);
225
226    let checksum = calculate_checksum(&full_msg_buf);
227    append_tag_value(&mut full_msg_buf, 10, &format!("{:03}", checksum));
228
229    Ok(full_msg_buf.freeze())
230}

233pub fn decode_fix_message(buffer: &mut BytesMut) -> Result<Option<FixMessage>, CodecError> {
234    // This is a very simplified parser. A real FIX parser is much more complex.
235    // It needs to handle partial messages, find SOH delimiters, extract tags and values robustly.
236
237    // Find 8=FIX...10=CHK<SOH>
238    let data_slice = buffer.as_ref();
239    let mut fields: std::collections::HashMap<u32, Vec<u8>> = std::collections::HashMap::new();
240    let mut current_pos = 0;
241
242    // Find BeginString (8=)
243    let tag8_pos = data_slice.windows(2).position(|w| w == b"8=").ok_or(CodecError::IncompleteMessage)?;
244    current_pos = tag8_pos;
245
246    let mut body_length: Option<u32> = None;
247    let mut msg_type: Option<String> = None;
248    let mut checksum_val: Option<u8> = None;
249    let mut header_map = std::collections::HashMap::new();
250    let mut body_map = std::collections::HashMap::new();
251
252    let mut last_field_end = current_pos;
253
254    loop {
255        if current_pos >= data_slice.len() { break; }
256        let remaining_slice = &data_slice[current_pos..];
257        let eq_pos = remaining_slice.iter().position(|&b| b == b'=').ok_or(CodecError::IncompleteMessage)?;
258        let tag_str = str::from_utf8(&remaining_slice[..eq_pos])?;
259        let tag: u32 = tag_str.parse().map_err(|_| CodecError::InvalidFormat(format!("Invalid tag: {}", tag_str)))?;
260
261        let soh_pos = remaining_slice[eq_pos+1..].iter().position(|&b| b == SOH).ok_or(CodecError::IncompleteMessage)?;
262        let value_slice = &remaining_slice[eq_pos+1 .. eq_pos+1+soh_pos];
263        let value_vec = value_slice.to_vec();
264
265        current_pos += eq_pos + 1 + soh_pos + 1;
266        last_field_end = current_pos;
267
268        match tag {
269            8 | 9 | 35 | 49 | 56 | 34 | 52 => { // Header tags
270                header_map.insert(tag, value_vec.clone());
271                if tag == 9 {
272                    body_length = Some(String::from_utf8(value_vec.clone())?.parse()?);
273                }
274                if tag == 35 {
275                    msg_type = Some(String::from_utf8(value_vec.clone())?);
276                }
277            }
278            10 => { // Trailer tag
279                checksum_val = Some(String::from_utf8(value_vec)?.parse()?);
280                break; // Checksum is the last field
281            }
282            _ => { // Body tags
283                body_map.insert(tag, value_vec);
284            }
285        }
286    }
287
288    let calculated_checksum_upto_tag10_start = calculate_checksum(&data_slice[tag8_pos..last_field_end - format!("10={:03}{}", checksum_val.unwrap_or(0), SOH as char).len()]);
289
290    if checksum_val.is_none() { return Err(CodecError::MissingField(10)); }
291    if calculated_checksum_upto_tag10_start != checksum_val.unwrap() {
292        return Err(CodecError::ChecksumMismatch { expected: checksum_val.unwrap(), actual: calculated_checksum_upto_tag10_start });
293    }
294
295    let header = FixHeader {
296        begin_string: String::from_utf8(header_map.remove(&8).ok_or(CodecError::MissingField(8))?)?,
297        body_length: body_length.ok_or(CodecError::MissingField(9))?,
298        msg_type: msg_type.clone().ok_or(CodecError::MissingField(35))?,
299        sender_comp_id: String::from_utf8(header_map.remove(&49).ok_or(CodecError::MissingField(49))?)?,
300        target_comp_id: String::from_utf8(header_map.remove(&56).ok_or(CodecError::MissingField(56))?)?,
301        msg_seq_num: String::from_utf8(header_map.remove(&34).ok_or(CodecError::MissingField(34))?)?.parse()?,
302        sending_time: String::from_utf8(header_map.remove(&52).ok_or(CodecError::MissingField(52))?)?.parse::<u64>()?,
303    };
304
305    let body = match msg_type.as_deref() {
306        Some("D") => FixMessageBody::NewOrderSingle(FixNewOrderSingle::decode_body(&mut body_map)?),
307        Some("A") => FixMessageBody::Logon(FixLogon::decode_body(&mut body_map)?),
308        // Add other message types here
309        Some(unknown_type) => return Err(CodecError::UnsupportedMessageType(unknown_type.to_string())),
310        None => return Err(CodecError::MissingField(35)),
311    };
312
313    buffer.split_to(last_field_end); // Consume the processed message from the buffer
314    Ok(Some(FixMessage { header, body }))
315}
